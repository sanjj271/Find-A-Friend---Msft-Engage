182      private String saveToInternalStorage(Bitmap bitmapImage){
183          ContextWrapper cw = new ContextWrapper(getApplicationContext());
184          // path to /data/data/yourapp/app_data/imageDir
185          File directory = cw.getDir("imageDir", Context.MODE_PRIVATE);
186          // Create imageDir
187          File mypath=new File(directory,"profile.jpg");
188  
189          FileOutputStream fos = null;
190          try {
191              fos = new FileOutputStream(mypath);
192              // Use the compress method on the BitMap object to write image to the OutputStream
193              bitmapImage.compress(Bitmap.CompressFormat.PNG, 100, fos);
194          } catch (Exception e) {
195              e.printStackTrace();
196          } finally {
197              try {
198                  fos.close();
199              } catch (IOException e) {
200                  e.printStackTrace();
201              }
202          }
203          return directory.getAbsolutePath();
204      }

235      private void detectAndFrame(final Bitmap imageBitmap) {
236          detectionProgressDialog = new ProgressDialog(this);
237          ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
238          imageBitmap.compress(Bitmap.CompressFormat.JPEG, 100, outputStream);
239          ByteArrayInputStream inputStream =
240                  new ByteArrayInputStream(outputStream.toByteArray());
241  
242          AsyncTask<InputStream, String, Face[]> detectTask =
243                  new AsyncTask<InputStream, String, Face[]>() {
244                      String exceptionMessage = "";
245  
246                      @Override
247                      protected Face[] doInBackground(InputStream... params) {
248                          try {
249                              publishProgress("Detecting...");
250                              Face[] result = faceServiceClient.detect(
251                                      params[0],
252                                      true,         // returnFaceId
253                                      false,        // returnFaceLandmarks
254                                      null          // returnFaceAttributes:
255                                      /* new FaceServiceClient.FaceAttributeType[] { 
256                                          FaceServiceClient.FaceAttributeType.Age, 
257                                          FaceServiceClient.FaceAttributeType.Gender } 
258                                      */
259                              );
260                              if (result == null){
261                                  publishProgress(
262                                          "Detection Finished. Nothing detected");
263                                  return null;
264                              }
265                              publishProgress(String.format(
266                                      "Detection Finished. %d face(s) detected",
267                                      result.length));
268                              return result;
269                          } catch (Exception e) {
270                              exceptionMessage = String.format(
271                                      "Detection failed: %s", e.getMessage());
272                              return null;
273                          }
274                      }
275  
276                      @Override
277                      protected void onPreExecute() {
278                          //TODO: show progress dialog
279                          detectionProgressDialog.show();
280                      }
281                      @Override
282                      protected void onProgressUpdate(String... progress) {
283                          //TODO: update progress
284                          detectionProgressDialog.setMessage(progress[0]);
285                      }
286                      @Override
287                      protected void onPostExecute(Face[] result) {
288  
289                          //TODO: update face frames
290                          detectionProgressDialog.dismiss();
291  
292                          if(!exceptionMessage.equals("")){
293                              showError(exceptionMessage);
294                          }
295                          if (result == null) return;
296                          matchPersonPhoto(result[0]);
297  
298  
299                      }
300                  };
301  
302          detectTask.execute(inputStream);
303      }

416      private void editProfile(){
417  
418          isEditmode=true;
419          editmode.setVisibility(View.VISIBLE);
420          viewmode.setVisibility(View.GONE);
421          editname.setText(viewname.getText().toString());
422          edithobbies.setText(viewhobbies.getText());
423          // initialize selected language array
424          selectedhobbies = new boolean[hobbies.length];
425          for (int i=0;i<hobbies.length;i++) {
426              if(viewhobbies.getText().toString().contains(hobbies[i])){
427                  selectedhobbies[i]=true;
428                  langList.add(i);
429              }
430          }
431          selectHobbies.setOnClickListener(new View.OnClickListener() {
432              @Override
433              public void onClick(View view) {
434  
435                  // Initialize alert dialog
436                  AlertDialog.Builder builder = new AlertDialog.Builder(Profile.this);
437  
438                  // set title
439                  builder.setTitle("Select your hobbies");
440  
441                  // set dialog non cancelable
442                  builder.setCancelable(false);
443  
444                  builder.setMultiChoiceItems(hobbies, selectedhobbies, new DialogInterface.OnMultiChoiceClickListener() {
445                      @Override
446                      public void onClick(DialogInterface dialogInterface, int i, boolean b) {
447                          // check condition
448                          if (b) {
449                              // when checkbox selected
450                              // Add position  in lang list
451                              langList.add(i);
452                              // Sort array list
453                              Collections.sort(langList);
454                          } else {
455                              // when checkbox unselected
456                              // Remove position from langList
457                              langList.remove(Integer.valueOf(i));
458                          }
459                      }
460                  });
461  
462                  builder.setPositiveButton("OK", new DialogInterface.OnClickListener() {
463                      @Override
464                      public void onClick(DialogInterface dialogInterface, int i) {
465                          // Initialize string builder
466                          StringBuilder stringBuilder = new StringBuilder();
467                          // use for loop
468                          for (int j = 0; j < langList.size(); j++) {
469                              // concat array value
470                              stringBuilder.append(hobbies[langList.get(j)]);
471                              // check condition
472                              if (j != langList.size() - 1) {
473                                  // When j value  not equal
474                                  // to lang list size - 1
475                                  // add comma
476                                  stringBuilder.append(", ");
477                              }
478                          }
479                          // set text on textView
480                          selectHobbies.setText(stringBuilder.toString());
481                      }
482                  });
483  
484                  builder.setNegativeButton("Cancel", new DialogInterface.OnClickListener() {
485                      @Override
486                      public void onClick(DialogInterface dialogInterface, int i) {
487                          // dismiss dialog
488                          dialogInterface.dismiss();
489                      }
490                  });
491                  builder.setNeutralButton("Clear All", new DialogInterface.OnClickListener() {
492                      @Override
493                      public void onClick(DialogInterface dialogInterface, int i) {
494                          // use for loop
495                          for (int j = 0; j < selectedhobbies.length; j++) {
496                              // remove all selection
497                              selectedhobbies[j] = false;
498                              // clear language list
499                              langList.clear();
500                              // clear text view value
501                              selectHobbies.setText("");
502  
503                          }
504                      }
505                  });
506                  // show dialog
507                  builder.show();
508              }
509          });
