188      private void addFaceToRegisterUser(final UUID personId) {
189          addingFace=new ProgressDialog(this);
190          Intent intent = getIntent();
191          Bitmap bitmap = (Bitmap) intent.getParcelableExtra("BitmapImage");
192          ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
193          bitmap.compress(Bitmap.CompressFormat.JPEG, 100, outputStream);
194          ByteArrayInputStream inputStream =
195                  new ByteArrayInputStream(outputStream.toByteArray());
196          AsyncTask<InputStream, String, AddPersistedFaceResult> addFaceTask =
197                  new AsyncTask<InputStream, String, AddPersistedFaceResult>() {
198                      String exceptionMessage = "";
199  
200                      @Override
201                      protected AddPersistedFaceResult doInBackground(InputStream... params) {
202                          try {
203                              publishProgress("Adding Face...");
204                              AddPersistedFaceResult result = faceServiceClient.addPersonFace("teleparadigm",personId,params[0],null,null);
205                              if (result == null){
206                                  publishProgress(
207                                          "Could not add face to person.");
208                                  return null;
209                              }
210  
211                              return result;
212                          } catch (Exception e) {
213                              exceptionMessage = String.format(
214                                      "Failed to add face: %s", e.getMessage());
215                              return null;
216                          }
217                      }
218  
219                      @Override
220                      protected void onPreExecute() {
221                          //TODO: show progress dialog
222                          addingFace.show();
223                      }
224                      @Override
225                      protected void onProgressUpdate(String... progress) {
226                          //TODO: update progress
227                          addingFace.setMessage(progress[0]);
228                      }
229                      @Override
230                      protected void onPostExecute(AddPersistedFaceResult result) {
231  
232                          //TODO: update face frames
233                          addingFace.dismiss();
234  
235                          if(!exceptionMessage.equals("")){
236                              showError(exceptionMessage);
237                          }

255                  };
