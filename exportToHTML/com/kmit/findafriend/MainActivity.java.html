52       private static Bitmap drawFaceRectanglesOnBitmap(
53               Bitmap originalBitmap, Face[] faces) {
54           Bitmap bitmap = originalBitmap.copy(Bitmap.Config.ARGB_8888, true);
55           Canvas canvas = new Canvas(bitmap);
56           Paint paint = new Paint();
57           paint.setAntiAlias(true);
58           paint.setStyle(Paint.Style.STROKE);
59           paint.setColor(Color.RED);
60           paint.setStrokeWidth(10);
61           if (faces != null) {
62               for (Face face : faces) {
63                   FaceRectangle faceRectangle = face.faceRectangle;
64                   canvas.drawRect(
65                           faceRectangle.left,
66                           faceRectangle.top,
67                           faceRectangle.left + faceRectangle.width,
68                           faceRectangle.top + faceRectangle.height,
69                           paint);
70               }
71           }
72           return bitmap;
73       }
